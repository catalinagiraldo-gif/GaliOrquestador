## 1. Diagnóstico — Por qué el estado actual no alcanza

### 1.1 Las señales son puramente reactivas — el diferenciador predictivo no existe visualmente

El valor más importante de Gali frente a cualquier IA genérica es que **ve el futuro comercial del usuario** usando datos de la red Dropi. Hoy el prototipo solo muestra alertas de problemas existentes. No hay ninguna pantalla donde el usuario vea: "en 14 días tu segmento de mascotas se satura" o "hay una ventana de mercado en Cali esta semana antes de que llegue la competencia".

Cita de `DesADIGali5.md §8`:
> "La distinción entre señal predictiva (oportunidad) y alerta reactiva (problema) **cambia el modelo mental del usuario**. Gali deja de ser un sistema de alarmas para convertirse en un estratega que ve el futuro."

### 1.2 El layout de dos columnas iguales no comunica la jerarquía correcta

El grid de Spec 4 (`grid-template-columns: 1fr 1fr`) trata señales y alertas como elementos de igual peso. La propuesta de Cambios10jun es un layout **maestro-detalle**: lista compacta a la izquierda, panel de contexto completo a la derecha. Este patrón permite mostrar toda la información de una señal sin saturar la lista.

### 1.3 Los botones del workspace-mode-bar no tienen destino

Los botones Operar / Lanzar / Medir en el mode-bar tienen `data-proto-skip`. El Spec 12 los activa:
- **Operar** → `/gali-v5/senales?filtro=alertas`
- **Lanzar** → `/gali-v5/proyectos/nuevo` pre-relleno con contexto de la señal seleccionada
- **Medir** → abre thread de Kronos en el panel Gali con datos de la señal

### 1.4 Sin badge vivo en navegación, las señales son invisibles entre sesiones

El usuario que lleva 2 horas en otra sección no sabe que Gali detectó una ventana de mercado. El badge del ítem "Señales" en el rail necesita dos conteos diferenciados que persisten mientras hay señales activas sin atender.

- Al seleccionar una señal: panel derecho hace `gali-fade-up` 200ms
- Al seleccionar una alerta crítica: panel derecho muestra el contexto + botón de acción urgente prominente

### C2 — Refactorizar `senales-page.component.ts` con estado de selección y navegación

**Archivo**: `src/app/pages/gali-v5/pages/senales/senales-page.component.ts`

```typescript
import { Component, signal, computed, inject, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { Router, ActivatedRoute } from '@angular/router';
import { TooltipModule } from 'primeng/tooltip';
import {
  MOCK_SENALES,
  MOCK_ALERTAS,
  GaliSignal,
  GaliAlerta,
} from '../../../../../mocks/gali-v5/senales.mock';
import { SenalDetalleComponent } from './senal-detalle/senal-detalle.component';

type FilterTab = 'todas' | 'senales' | 'alertas' | 'completadas';
type SelectedItem = { kind: 'senal'; data: GaliSignal } | { kind: 'alerta'; data: GaliAlerta };

@Component({
  selector: 'app-senales-page',
  standalone: true,
  imports: [CommonModule, TooltipModule, SenalDetalleComponent],
  templateUrl: './senales-page.component.html',
  styleUrls: ['./senales-page.component.scss'],
})
export class SenalesPageComponent implements OnInit {
  private router = inject(Router);
  private route = inject(ActivatedRoute);

  readonly senales = MOCK_SENALES;
  readonly alertas = MOCK_ALERTAS;

  activeFilter = signal<FilterTab>('todas');
  selectedId = signal<string | null>(null);

  // ──────────────────────────────────────────────────────────
  // Computed: listas visibles según filtro activo
  // ──────────────────────────────────────────────────────────

  senalesActivas = computed(() => this.senales.filter(s => s.tipo !== 'completed'));
  alertasActivas = computed(() => this.alertas);

  senalesVisibles = computed(() => {
    if (this.activeFilter() === 'alertas') return [];
    if (this.activeFilter() === 'completadas') return this.senales.filter(s => s.tipo === 'completed');
    return this.senales.filter(s => s.tipo !== 'completed');
  });

  alertasVisibles = computed(() => {
    if (this.activeFilter() === 'senales' || this.activeFilter() === 'completadas') return [];
    return this.alertas;
  });

  // ──────────────────────────────────────────────────────────
  // Computed: ítem seleccionado
  // ──────────────────────────────────────────────────────────

  selectedItem = computed<SelectedItem | null>(() => {
    const id = this.selectedId();
    if (!id) return null;
    const senal = this.senalesVisibles().find(s => s.id === id);
    if (senal) return { kind: 'senal', data: senal };
    const alerta = this.alertasVisibles().find(a => a.id === id);
    if (alerta) return { kind: 'alerta', data: alerta };
    return null;
  });

  isAlertSelected = computed(() => this.selectedItem()?.kind === 'alerta');

  // ──────────────────────────────────────────────────────────
  // Tabs de filtro con conteos reactivos
  // ──────────────────────────────────────────────────────────

  filterTabs = [
    { value: 'todas' as FilterTab, label: 'Todas', count: computed(() => this.senalesActivas().length + this.alertasActivas().length) },
    { value: 'senales' as FilterTab, label: 'Señales', count: computed(() => this.senalesActivas().length) },
    { value: 'alertas' as FilterTab, label: 'Alertas', count: computed(() => this.alertasActivas().length) },
    { value: 'completadas' as FilterTab, label: 'Completadas', count: computed(() => this.senales.filter(s => s.tipo === 'completed').length) },
  ];

  // ──────────────────────────────────────────────────────────
  // Modal de confirmación (Spec 4 Fase 2)
  // ──────────────────────────────────────────────────────────

  showConfirmModal = signal(false);
  confirmConfig = signal<any | null>(null);

  // ──────────────────────────────────────────────────────────
  // Lifecycle
  // ──────────────────────────────────────────────────────────

  ngOnInit() {
    const params = this.route.snapshot.queryParamMap;
    const filtroParam = params.get('filtro') as FilterTab | null;
    const signalId = params.get('signalId');

    if (filtroParam && ['todas', 'senales', 'alertas', 'completadas'].includes(filtroParam)) {
      this.activeFilter.set(filtroParam);
    }

    if (signalId) {
      this.selectedId.set(signalId);
    } else {
      // Seleccionar el primer ítem por defecto
      const primeraSenal = this.senalesVisibles()[0];
      const primeraAlerta = this.alertasVisibles()[0];
      const defaultId = primeraSenal?.id ?? primeraAlerta?.id ?? null;
      this.selectedId.set(defaultId);
    }
  }

  // ──────────────────────────────────────────────────────────
  // Handlers de selección
  // ──────────────────────────────────────────────────────────

  setFilter(f: FilterTab) {
    this.activeFilter.set(f);
    // Resetear selección al cambiar filtro
    const primeraSenal = this.senalesVisibles()[0];
    const primeraAlerta = this.alertasVisibles()[0];
    this.selectedId.set(primeraSenal?.id ?? primeraAlerta?.id ?? null);
  }

  selectSignal(id: string) { this.selectedId.set(id); }
  selectAlert(id: string) { this.selectedId.set(id); }

  // ──────────────────────────────────────────────────────────
  // Handlers de CTA desde el panel detalle
  // ──────────────────────────────────────────────────────────

  onCtaPrimario(item: SelectedItem) {
    if (item.kind === 'alerta' && item.data.tipo === 'critical') {
      this.abrirModalConfirmacion(item.data);
    } else {
      // Señales y alertas no críticas: acción directa (log en fase 1)
      console.log('CTA primario:', item.data.id);
    }
  }

  onCtaSecundario(item: SelectedItem) {
    console.log('CTA secundario:', item.data.id);
  }

  onLanzar() {
    const selected = this.selectedItem();
    if (selected) {
      this.router.navigate(['/gali-v5/proyectos/nuevo'], {
        queryParams: { signalId: selected.data.id }
      });
    } else {
      this.router.navigate(['/gali-v5/proyectos/nuevo']);
    }
  }

  onOperar() {
    this.activeFilter.set('alertas');
    const primeraAlerta = this.alertasVisibles()[0];
    if (primeraAlerta) this.selectedId.set(primeraAlerta.id);
  }

  onMedir() {
    // Abre thread de Kronos en panel Gali con datos de la señal seleccionada
    // Por ahora navega al Hub con el panel Gali abierto
    const selected = this.selectedItem();
    this.router.navigate(['/gali-v5/home'], {
      queryParams: { openGali: 'kronos', signalId: selected?.data.id ?? null }
    });
  }

  onIgnorar(item: SelectedItem) {
    // En prototipo: simplemente deseleccionar y log
    console.log('Señal ignorada:', item.data.id);
    this.selectedId.set(null);
  }

  onGuardarDespues(item: SelectedItem) {
    console.log('Guardado para después:', item.data.id);
  }

  // ──────────────────────────────────────────────────────────
  // Helpers de UI
  // ──────────────────────────────────────────────────────────

  getTipoIcon(tipo: string): string {
    const icons: Record<string, string> = {
      scale: '⚡',
      trend: '🔭',
      opportunity: '💎',
      risk: '⚠️',
      completed: '✅',
    };
    return icons[tipo] ?? '•';
  }

  private abrirModalConfirmacion(alerta: GaliAlerta) {
    const titulos: Record<string, string> = {
      'alt-001': 'Cambiar transportadora',
      'alt-002': 'Reconectar integración',
      'alt-003': 'Activar creative alternativo',
      'alt-004': 'Activar pausa automática',
      'alt-005': 'Reasignar pedidos urgentes',
    };
    this.confirmConfig.set({
      titulo: titulos[alerta.id] ?? 'Confirmar acción',
      descripcion: alerta.descripcion,
      impacto: alerta.impacto,
      iconoAgente: alerta.agente,
      itemsAfectados: alerta.pedidosAfectados ?? undefined,
      labelItems: alerta.pedidosAfectados ? 'pedidos' : undefined,
      ctaConfirmar: alerta.ctaPrincipal,
      ctaCancelar: 'Cancelar',
      onConfirm: () => {
        this.showConfirmModal.set(false);
        console.log('Acción confirmada para alerta', alerta.id);
      },
      onCancel: () => this.showConfirmModal.set(false),
    });
    this.showConfirmModal.set(true);
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter, computed, signal } from '@angular/core';
import { CommonModule } from '@angular/common';
import { GaliSignal, GaliAlerta } from '../../../../../../mocks/gali-v5/senales.mock';

type SelectedItem = { kind: 'senal'; data: GaliSignal } | { kind: 'alerta'; data: GaliAlerta };

@Component({
  selector: 'app-senal-detalle',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './senal-detalle.component.html',
  styleUrls: ['./senal-detalle.component.scss'],
})
export class SenalDetalleComponent {
  @Input() item!: SelectedItem;
  @Input() isAlerta = false;

  @Output() ctaPrimario = new EventEmitter<SelectedItem>();
  @Output() ctaSecundario = new EventEmitter<SelectedItem>();
  @Output() ignorar = new EventEmitter<SelectedItem>();
  @Output() guardarDespues = new EventEmitter<SelectedItem>();

  get senal(): GaliSignal | null {
    return this.item?.kind === 'senal' ? this.item.data : null;
  }

  get alerta(): GaliAlerta | null {
    return this.item?.kind === 'alerta' ? this.item.data : null;
  }

  get windowClass(): string {
    if (!this.senal) return '';
    if (this.senal.ventanaDias <= 6) return 'urgent';
    if (this.senal.ventanaDias <= 9) return 'soon';
    return 'ok';
  }

  get windowColor(): string {
    if (!this.senal) return '';
    if (this.senal.ventanaDias <= 6) return 'var(--red-600)';
    if (this.senal.ventanaDias <= 9) return 'var(--amber-600)';
    return 'var(--green-600)';
  }
}
```
